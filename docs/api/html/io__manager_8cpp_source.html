23ProjectFile IOManager::importProject(const std::string& file_path)
24{
25 std::ifstream inputFile(file_path);
26 if (!inputFile.is_open()) {
27 throw std::runtime_error("Failed to open project file: " + file_path);
28 }
29
30 json projectJson;
31 inputFile >> projectJson;
32 inputFile.close();
33
34 const int width = projectJson.at("width").get<int>();
35 const int height = projectJson.at("height").get<int>();
36
37 ProjectFile project(width, height);
38
39 // Import layers
40 if (projectJson.contains("layers")) {
41 for (const auto& layerJson : projectJson.at("layers")) {
42 auto layer = std::make_shared<Layer>(width, height);
43
44 // Restore layer properties
45 layer->set_name(layerJson.at("name").get<std::string>());
46 layer->set_visible(layerJson.at("visible").get<bool>());
47 layer->set_opacity(layerJson.at("opacity").get<float>());
48 layer->set_blend_mode(
49 string_to_blend_mode(layerJson.at("blend_mode").get<std::string>()));
50
51 // Restore layer data
52 const std::vector<uint8_t> layerData = layerJson.at("data").get<std::vector<uint8_t>>();
53 if (layerData.size() == layer->data().size()) {
54 layer->data() = layerData;
55 } else {
56 throw std::runtime_error("Layer data size mismatch during import");
57 }
58
59 project.add_layer(); // This adds to the layer stack
60 }
61 }
62
63 return project;
64}

66bool IOManager::exportProject(const ProjectFile& project, const std::string& file_path)
67{
68 try {
69 json projectJson;
70
71 projectJson["width"] = project.width();
72 projectJson["height"] = project.height();
73
74 json layersJson = json::array();
75
76 // Export each layer
77 for (const auto& layer : project.layers()) {
78 json layerJson;
79 layerJson["name"] = layer->name();
80 layerJson["visible"] = layer->visible();
81 layerJson["opacity"] = layer->opacity();
82 layerJson["blend_mode"] = blend_mode_to_string(layer->blend_mode());
83 layerJson["width"] = layer->width();
84 layerJson["height"] = layer->height();
85 layerJson["data"] = layer->data();
86
87 layersJson.push_back(layerJson);
88 }
89
90 projectJson["layers"] = layersJson;
91
92 std::ofstream outputFile(file_path);
93 if (!outputFile.is_open()) {
94 return false;
95 }
96
97 outputFile << projectJson.dump(4); // Pretty-print with 4-space indent
98 outputFile.close();
99
100 return true;
101 } catch (const std::exception& e) {
102 std::cerr << "Error exporting project: " << e.what() << '\n';
103 return false;
104 }
105}
