80void ErrorHandler::ReportError(const ErrorInfo& error)
81{
82 const std::scoped_lock lock(mutex_);
83
84 // Add to history
85 error_history_.push_back(error);
86
87 // Log the error
88 LogError(error);
89
90 // Invoke callbacks for this error category
91 for (const auto& [category, callback] : callbacks_) {
92 if (category == ErrorCategory::Unknown || category == error.GetCategory()) {
93 callback(error);
94 }
95 }
96
97 // Handle based on severity
98 switch (error.GetSeverity()) {
99 case ErrorSeverity::Info:
100 case ErrorSeverity::Warning:
101 // Just log, don't show dialog
102 std::cout << error.ToString() << '\n';
103 break;
104
105 case ErrorSeverity::Recoverable:
106 // Try to recover
107 std::cerr << error.ToString() << '\n';
108 if (!AttemptRecovery(error)) {
109 // Recovery failed, show error dialog if enabled
110 if (show_error_dialogs_) {
111 ShowErrorDialog(error);
112 }
113 }
114 break;
115
116 case ErrorSeverity::Fatal:
117 // Fatal errors are handled separately
118 std::cerr << "FATAL: " << error.ToString() << '\n';
119 // Note: HandleFatalError should be called directly for fatal errors
120 break;
121 }
122}

161bool ErrorHandler::AttemptRecovery(const ErrorInfo& error)
162{
163 const std::scoped_lock lock(mutex_);
164
165 // Try recovery handlers for this category

235void ErrorHandler::LogError(const ErrorInfo& error)
236{
237 // Get current timestamp
238 auto now = std::time(nullptr);
239 std::tm tm{};
240 localtime_s(&tm, &now);
241
242 std::ostringstream oss;
243 oss << std::put_time(&tm, "%Y-%m-%d %H:%M:%S");
244 oss << " | " << error.ToString() << '\n';
245
246 // Write to log file
247 std::ofstream logFile(log_file_path_, std::ios::app);
248 if (logFile.is_open()) {
249 logFile << oss.str();
250 logFile.flush();
251 }
252}

254void ErrorHandler::ShowErrorDialog(const ErrorInfo& error)
255{
256 // NOLINTNEXTLINE(readability-implicit-bool-conversion)
257 if (!QApplication::instance()) {
258 return; // Can't show dialog without QApplication
259 }
260
261 QString title;
262 QMessageBox::Icon icon = QMessageBox::Information;
263
264 switch (error.GetSeverity()) {
265 case ErrorSeverity::Warning:
266 title = "Warning";
267 icon = QMessageBox::Warning;
268 break;
269 case ErrorSeverity::Recoverable:
270 title = "Error";
271 icon = QMessageBox::Critical;
272 break;
273 case ErrorSeverity::Fatal:
274 title = "Fatal Error";
275 icon = QMessageBox::Critical;
276 break;
277 default:
278 title = "Information";
279 break;
280 }
281
282 QMessageBox msgBox;
283 msgBox.setIcon(icon);
284 msgBox.setWindowTitle(title);
285 msgBox.setText(QString::fromStdString(error.ToString()));
286 msgBox.setStandardButtons(QMessageBox::Ok);
287 msgBox.exec();
288}
