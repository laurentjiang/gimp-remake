24DebugHud::DebugHud(QWidget* parent) : QWidget(parent)
25{
26 setupUi();
27
28 updateTimer_ = new QTimer(this);
29 connect(updateTimer_, &QTimer::timeout, this, &DebugHud::updateStats);
30 updateTimer_->start(500);
31
32 mousePosSub_ = EventBus::instance().subscribe<MousePositionChangedEvent>(
33 [this](const MousePositionChangedEvent& event) {
34 updateMousePosition(event.canvasX, event.canvasY);
35 });
36
37 zoomSub_ = EventBus::instance().subscribe<CanvasViewChangedEvent>(
38 [this](const CanvasViewChangedEvent& event) {
39 zoomLevel_ = event.zoomLevel;
40 zoomLabel_->setText(QString("Zoom: %1%").arg(static_cast<int>(zoomLevel_ * 100)));
41 });
42
43 setAttribute(Qt::WA_TransparentForMouseEvents);
44}

120void DebugHud::updateStats()
121{
122 const double fps = calculateFps();
123 fpsLabel_->setText(QString("FPS: %1").arg(fps, 0, 'f', 1));
124
125 const std::size_t memMb = getMemoryUsage() / static_cast<std::size_t>(1024 * 1024);
126 memoryLabel_->setText(QString("Memory: %1 MB").arg(memMb));
127
128 if (document_) {
129 const std::size_t layerCount = document_->layers().count();
130 layerCountLabel_->setText(QString("Layers: %1").arg(layerCount));
131
132 canvasSizeLabel_->setText(
133 QString("Canvas: %1 x %2").arg(document_->width()).arg(document_->height()));
134 }
135}

137double DebugHud::calculateFps()
138{
139 if (frameTimes_.size() < 2) {
140 return 0.0;
141 }
142
143 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(frameTimes_.back() -
144 frameTimes_.front());
145
146 if (duration.count() == 0) {
147 return 0.0;
148 }
149
150 return static_cast<double>(frameTimes_.size() - 1) * 1000.0 /
151 static_cast<double>(duration.count());
152}

154std::size_t DebugHud::getMemoryUsage()
155{
156#ifdef _WIN32
157 PROCESS_MEMORY_COUNTERS_EX pmc;
158 if (GetProcessMemoryInfo(
159 GetCurrentProcess(), reinterpret_cast<PROCESS_MEMORY_COUNTERS*>(&pmc), sizeof(pmc))) {
160 return static_cast<std::size_t>(pmc.WorkingSetSize);
161 }
162#endif
163 return 0;
164}
