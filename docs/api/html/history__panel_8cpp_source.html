77void HistoryPanel::addEntry(const std::string& actionName)
78{
79 if (currentIndex_ >= 0 && currentIndex_ < static_cast<int>(entries_.size()) - 1) {
80 entries_.erase(entries_.begin() + currentIndex_ + 1, entries_.end());
81 }
82
83 // NOLINTNEXTLINE(modernize-use-designated-initializers)
84 entries_.push_back({actionName, false});
85 currentIndex_ = static_cast<int>(entries_.size()) - 1;
86 refreshList();
87}

96void HistoryPanel::refreshList()
97{
98 historyList_->clear();
99
100 for (std::size_t i = 0; i < entries_.size(); ++i) {
101 auto* item = new QListWidgetItem(QString::fromStdString(entries_[i].actionName));
102 item->setData(Qt::UserRole, static_cast<int>(i));
103
104 // NOLINTNEXTLINE(modernize-use-integer-sign-comparison)
105 if (static_cast<int>(i) > currentIndex_) {
106 item->setForeground(Qt::gray);
107 // NOLINTNEXTLINE(modernize-use-integer-sign-comparison)
108 } else if (static_cast<int>(i) == currentIndex_) {
109 item->setBackground(QColor(200, 220, 255));
110 }
111
112 historyList_->addItem(item);
113 }
114
115 if (currentIndex_ >= 0 && currentIndex_ < historyList_->count()) {
116 historyList_->setCurrentRow(currentIndex_);
117 }
118
119 undoButton_->setEnabled(currentIndex_ > 0);
120 redoButton_->setEnabled(currentIndex_ < static_cast<int>(entries_.size()) - 1);
121}

123void HistoryPanel::onItemClicked(QListWidgetItem* item)
124{
125 const int targetIndex = item->data(Qt::UserRole).toInt();
126 if (targetIndex != currentIndex_) {
127 currentIndex_ = targetIndex;
128 refreshList();
129 emit historyJumpRequested(currentIndex_);
